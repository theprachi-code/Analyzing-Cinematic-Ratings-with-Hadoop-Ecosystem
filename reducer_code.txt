#!/usr/bin/env python
import sys

current_movie = None
total = 0
count = 0

for line in sys.stdin:
    # Parse the input we got from mapper.py
    movie, rating = line.strip().split('\t')
    rating = float(rating)

    # If we are still processing the same movie, accumulate the ratings
    if current_movie == movie:
        total += rating
        count += 1
    else:
        # If the movie changes, output the average for the previous movie
        if current_movie:
            print("%s\t%s" % (current_movie, total/count))
        
        # Reset the trackers for the new movie
        current_movie = movie
        total = rating
        count = 1

# Don't forget to output the final movie's average after the loop ends
if current_movie:
    print("%s\t%s" % (current_movie, total/count))
